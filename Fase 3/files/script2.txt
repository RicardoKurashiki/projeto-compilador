{
    int x;
    int y;
    int buttonStatus;

    pinMode 2, 0;
    pinMode 3, 1;

    x = 20;
    y = 10 + 20;

    while (x <= y)
    {
        buttonStatus = digitalRead 2;
        if (buttonStatus == 0)
        {
            digitalWrite 3, 1;
        }
        else
        {
            digitalWrite 3, 0;
            x = x + 1;
        }
    }

    digitalWrite 3,0;
}
