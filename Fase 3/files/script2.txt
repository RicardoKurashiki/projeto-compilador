{
    int x;
    int y;

    x = 20;
    y = 10 + 24;
    int buttonStatus;
    pinMode 2, 0;
    pinMode 3, 1;

    while (x >= y)
    {
        buttonStatus = 1 + 2;
        if (buttonStatus == 0)
        {
            digitalWrite 3,1;
        }
        else
        {
            digitalWrite 3,0;
        }
    }
}
