{
    int x;
    int y;
    int buttonStatus;

    pinMode 2, 0;
    pinMode 13, 1;

    x = 20;
    y = 20;

    while (x == y)
    {
        buttonStatus = digitalRead 2;
        if (buttonStatus == 0)
        {
            digitalWrite 13, 0;
        }
        else
        {
            digitalWrite 13, 1;
        }
    }

    digitalWrite 13, 0;
}
